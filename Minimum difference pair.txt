class Solution
{
    public int minimum_difference(int[] nums)
    {
        // code here
        Arrays.sort(nums);
        int min = nums[nums.length-1];
        //System.out.println("min--> "+min);
        
        for(int i=1;i<nums.length;i++)
        {
            if(nums[i]-nums[i-1] < min) 
                min = nums[i]-nums[i-1];
                 //System.out.println("min inside--> "+min);
        }
        
        return min;
    }
}
