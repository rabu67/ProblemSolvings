//User function Template for Java

class Solution
{
    //Function to return sum of upper and lower triangles of a matrix.
    static ArrayList<Integer> sumTriangles(int matrix[][], int n)
    {
        // code here
        ArrayList<Integer> result = new ArrayList<>();
        
        int usum = 0;
        int lsum = 0;
        
        for(int i=0;i<n;i++) 
        {
            for(int j=0;j<n;j++) 
            {
                if(i==j) {
                    usum+=matrix[i][j];
                    lsum+=matrix[i][j];
                }
                
                else if(i < j) 
                {
                    usum+=matrix[i][j];
                }
                
                else 
                {
                    lsum+=matrix[i][j];
                }
            }
        }
        
        result.add(usum);
        result.add(lsum);
        return result;
    }
}
