class Solution 
{
    //Function to reverse words in a given string.
    String reverseWords(String S)
    {
        // code here 
        String[] arr = S.split("\\.");
        String result="";
        int arrSize = arr.length;
        for(int i=arrSize-1;i>=0;i--) {
            result+= arr[i]+"."; 
        }
        result = result.substring(0, result.length()-1);
        return result;
    }
}
